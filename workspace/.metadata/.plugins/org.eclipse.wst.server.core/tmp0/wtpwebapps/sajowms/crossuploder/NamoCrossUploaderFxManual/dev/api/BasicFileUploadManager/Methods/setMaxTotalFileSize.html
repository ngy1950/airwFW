setMaxTotalFileSize

파일 추가시 허용할 전체 파일의 최대 크기를 설정합니다.

 

Syntax

    void setMaxTotalFileSize ( Number maxTotalFileSize )

Parameters

maxTotalFileSize

[in] 파일 추가시 허용할 전체 파일의 최대 크기입니다.

 

Return Values

 

Remarks

 

Sample Codes

Javascript

/**
* NamoCrossUploader의 Manager 객체 생성 완료 시 호출됩니다.
*/
var onCreationCompleteCu = function () {
    cuManager = document.getElementById("crossUploadManager");
    // upload url 설정
    cuManager.setUploadURL("http://218.153.118.35:8080/NamoCrossUploaderFxSamples/Upload/LimitFileUpload/UploadProcess.jsp");
    // 허용할 파일 확장자 설정
    var imageFilter = { "desc": "이미지 파일(jpg, jpeg)", "ext": "*.jpg;*.jpeg" };
    var textFilter = { "desc": "텍스트 파일(txt)", "ext": "*.txt" };
    var allFilter = { "desc": "모든 파일", "ext": "*.*" };
    cuManager.setFileFilter([imageFilter, textFilter, allFilter]);
    // 허용하지 않을 파일 포맷 기준으로 File Filter를 설정하시려면 setFileFilter 메소드의 두번째 파라미터에 true를 입력해 주십시오.
    // cuManager.setFileFilter([imageFilter, textFilter, allFilter], true);
    // 파일 개수 제한
    cuManager.setMaxFileCount(3);
    
    // 전체 파일 크기를 10MB로 제한
    cuManager.setMaxTotalFileSize(1024 * 1024 * 10);
    
    // 개별 파일 크기를 10KB로 제한
    cuManager.setMaxFileSize(1024 * 10);
}
